% Project-thesis preamble.
% Mirrors ../preamble.tex but replaces the legacy `algorithmic` package with
% `algpseudocode` (from algorithmicx), which is what the markdown algorithm
% blocks actually use (\Require, \Ensure, \State, \For, ...).

% Force a page break after the table of contents
\AtBeginDocument{\let\oldtableofcontents\tableofcontents
\renewcommand{\tableofcontents}{\oldtableofcontents\newpage}}

% Prevent figures from floating past section boundaries
\usepackage[section]{placeins}

% Force figure placement to be near source text
\usepackage{float}
% Make every figure use [H] (no floating). With htbp, LaTeX may commit a
% figure to the bottom of the current page when a preceding longtable has
% eaten most of the vertical space, leaving the scaled image to overflow
% into the footer. [H] sidesteps the float algorithm: the figure either
% fits at the source position or LaTeX page-breaks before it.
\floatplacement{figure}{H}

% Relax LaTeX float-placement fractions so the float algorithm is willing to
% defer a figure to its own page rather than committing it to a bottom slot
% that the scaled image will then overflow. Symptom this fixes: a figure that
% follows a multi-row pandoc longtable lands at the bottom of the table's last
% page, then triggers `Overfull \vbox ... while \output is active` because the
% remaining vertical space is smaller than the figure. Defaults are too strict
% (textfraction=0.2, bottomfraction=0.3, floatpagefraction=0.5), which makes
% LaTeX prefer cramming over deferring; loosening them lets it pick a clean
% float page when the in-line slot won't fit.
\renewcommand{\textfraction}{0.05}
\renewcommand{\topfraction}{0.95}
\renewcommand{\bottomfraction}{0.95}
\renewcommand{\floatpagefraction}{0.5}

% Algorithm typesetting (algorithmicx style — provides \Require/\Ensure/\State/...)
\usepackage{algorithm}
\usepackage{algpseudocode}
% \ParFor / \EndParFor are not built-in; define them as styled \For variants.
\algnewcommand\algorithmicparfor{\textbf{parallel for}}
\algnewcommand\algorithmicparforend{\textbf{end\ parallel for}}
\algdef{SE}[PARFOR]{ParFor}{EndParFor}[1]%
  {\algorithmicparfor\ #1\ \algorithmicdo}%
  {\algorithmicparforend}

% Indent ALL first lines, including first paragraph after headings
\usepackage{indentfirst}

% Prevent Pandoc's parskip package from removing indentation
\usepackage{etoolbox}
\usepackage{adjustbox}

% Robust column sizing for wide tables (used by raw-LaTeX tabularx tables
% where pandoc's longtable auto-sizing produces lopsided columns).
\usepackage{tabularx}
\usepackage{booktabs}

% --- Table fixes: fit wide Pandoc longtables to page width ---
% Shrink table font, tighten column padding, and centre.
\AtBeginEnvironment{longtable}{%
  \footnotesize%
  \setlength{\tabcolsep}{3pt}%
  \setlength{\LTleft}{0pt plus 1fill}%
  \setlength{\LTright}{0pt plus 1fill}%
}
\AtEndPreamble{%
  \setlength{\parindent}{1.25cm}%
  \setlength{\parskip}{0pt}%
}
\AtBeginDocument{%
  \setlength{\parindent}{1.25cm}%
  \setlength{\parskip}{0pt}%
}

% Redefine pandocbounded to leave room for caption + spacing
% Default uses \textheight which doesn't account for caption/floatsep
\makeatletter
\renewcommand*\pandocbounded[1]{%
  \sbox\pandoc@box{#1}%
  \Gscale@div\@tempa{0.75\textheight}{\dimexpr\ht\pandoc@box+\dp\pandoc@box\relax}%
  \Gscale@div\@tempb{\linewidth}{\wd\pandoc@box}%
  \ifdim\@tempb\p@<\@tempa\p@\let\@tempa\@tempb\fi
  \ifdim\@tempa\p@<\p@\scalebox{\@tempa}{\usebox\pandoc@box}%
  \else\usebox{\pandoc@box}%
  \fi
}
\makeatother

% Heading styles: all at normal text size, differentiated by boldness only
% Indented like paragraph first lines (1.25cm)
\usepackage{titlesec}
\titleformat{\section}[block]{\normalfont\normalsize\bfseries\filcenter}{\thesection}{1em}{}
\titleformat{\subsection}{\normalfont\normalsize\bfseries}{\thesubsection}{1em}{}
\titleformat{\subsubsection}{\normalfont\normalsize\mdseries\itshape}{\thesubsubsection}{1em}{}
\titlespacing*{\section}{0pt}{1.5ex plus 0.5ex}{1ex plus 0.3ex}
\titlespacing*{\subsection}{\parindent}{1.5ex plus 0.5ex}{0.8ex plus 0.2ex}
\titlespacing*{\subsubsection}{\parindent}{1ex plus 0.3ex}{0.5ex plus 0.2ex}

% Page numbers centered in footer (GOST)
\usepackage{fancyhdr}
\pagestyle{fancy}
\fancyhf{}
\fancyfoot[C]{\thepage}
\renewcommand{\headrulewidth}{0pt}
\fancypagestyle{plain}{\fancyhf{}\fancyfoot[C]{\thepage}\renewcommand{\headrulewidth}{0pt}}
